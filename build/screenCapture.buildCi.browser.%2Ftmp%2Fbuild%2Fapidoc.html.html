module ratelimiter.index.prototype

function ratelimiter.index.prototype.get (fn)

  • description and source-code
    get = function (fn) {
  var db = this.db;
  var duration = this.duration;
  var key = this.key;
  var max = this.max;
  var now = microtime.now();
  var start = now - duration * 1000;

  db.multi()
    .zremrangebyscore([key, 0, start])
    .zcard([key])
    .zadd([key, now, now])
    .zrange([key, 0, 0])
    .pexpire([key, duration])
    .exec(function (err, res) {
      if (err) return fn(err);
      var count = parseInt(Array.isArray(res[0]) ? res[1][1] : res[1]);
      var oldest = parseInt(Array.isArray(res[0]) ? res[3][1] : res[3]);
      fn(null, {
        remaining: count < max ? max - count : 0,
        reset: Math.floor((oldest + duration * 1000) / 1000000),
        total: max
      });
    });
}
